prevent duplicate email contact

trigger contacttrigger on contact(before insert){
set<string> conemail = new set<string>();
for(contact con : trigger.new){
 if(con.email != null){
  conemail.add(con.email);
  }
 }
list<contact> conlist = [select email from contact where email in :conemail];

for(contact con : trigger.new){
 for(contact existingcon : conlist){
  if(con.email != null && con.email == existingcon.email){
    con.addError('Error msg');
   }
  }
 }
}
